```py
def DFS(graph, start):
    visited = {node : False for node in graph}
    s = [start]
    visited[start] = True
    while s:
        cur_node = s.pop()
        print(cur_node)
        for node in graph[cur_node]:
            if not visited[node]:
                visited[node] = True
                s.append(node)
        print("next : ", end="")
        for e in s:
            print(e, end=" ")
        print()
DFS(graph, 'A')
```
