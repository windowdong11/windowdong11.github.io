        # path-compression
        parentOf[node] = Find(parentOf[node])
    return parentOf[node]

def Union (n1, n2):
    p1 = Find(n1)
    p2 = Find(n2)

        parentOf[p2] = p1
    else:
        rank[p1] += 1
        parentOf[p2] = p1
